15.1 服务器权威性设计原理

15.1.0 核心概念

服务器权威性（Server Authority）是网络游戏安全的根本原则，其核心思想是：永远不要信任客户端。这意味着所有关键游戏逻辑的最终裁决权必须在服务器端，客户端仅仅是预测和展示。

为什么客户端不可信

客户端运行在用户设备上，理论上用户拥有完全控制权：

客户端可被攻击的方式：
├── 内存修改
│   └── 修改生命值、弹药、金币等数值
├── 网络层拦截
│   └── 篡改数据包，伪造请求
├── 自动化工具
│   └── 脚本、宏、外挂程序
├── 逆向工程
│   └── 分析协议，伪造客户端
└── 调试器附加
    └── 暂停游戏，单步执行，修改状态

网络游戏的常见攻击向量

攻击类型实现方式危害程度防护难度
速度修改修改移动速度数值中等易
透视修改渲染逻辑显示隐藏敌人高难
自动瞄准自动化输入模拟高中等
伤害修改修改武器伤害输出严重易
位置瞬移伪造位置数据包严重中等
无敌模式修改受伤逻辑严重易

服务器权威性的工作原理

客户端请求 → 服务器验证 → 状态更新 → 广播给所有客户端
     ↑                              ↓
   预测显示                    权威状态同步
  1. 客户端发送请求：玩家输入被发送到服务器
  2. 服务器严格验证：检查请求是否合法（位置、伤害、冷却等）
  3. 服务器执行逻辑：只有验证通过才更新游戏状态
  4. 广播状态更新：将权威状态同步给所有客户端
  5. 客户端 reconciliations：如有差异，客户端平滑过渡至服务器状态

15.1.1 安全与性能的权衡

服务器权威性带来了额外的网络延迟：

架构模式延迟安全性适用场景
纯客户端低无纯单机游戏
客户端预测+服务器权威中高竞技游戏
完全服务器模拟高最高策略游戏、回合制

优化策略：

  • 客户端预测：本地立即响应，等待服务器确认
  • 服务器调和：服务器状态与客户端预测 reconciliations
  • 乐观更新：客户端假设操作成功，失败时回滚

15.1.2 使用场景分析

适用场景

服务器权威性验证适用于以下游戏类型：

  • 竞技游戏：排行榜、排名、段位系统必须严格公平
  • 付费游戏：涉及真实货币交易，必须防止作弊获利
  • 多人在线：影响其他玩家体验的游戏行为需要验证
  • 持续运营：长期运营的游戏需要健康生态

不适用场景

以下场景可以适当放宽验证：

  • 纯合作PVE：玩家之间无竞争，作弊仅影响自己
  • 局域网游戏：信任环境，作弊影响有限
  • 单机游戏：完全离线，无服务器验证可能
  • 演示版本：临时体验，不涉及长期数据

15.1 服务器权威性验证

服务器权威性（Server Authority）是网络安全的核心原则，意味着所有关键游戏逻辑的最终裁决权必须在服务器端。

15.1.1 服务器校验逻辑

服务器校验确保客户端发送的所有操作都经过严格验证。

using Godot;
using System;
using System.Collections.Generic;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 服务器权威验证管理器
    /// </summary>
    public class ServerAuthorityValidator : Node
    {
        // 服务器游戏状态
        private ServerGameState _serverState;

        // 玩家状态缓存
        private Dictionary<int, PlayerServerState> _playerStates = new();

        // 验证配置
        [Export] public float MaxPositionDelta { get; set; } = 5.0f;
        [Export] public float MaxSpeed { get; set; } = 20.0f;
        [Export] public int MaxHealth { get; set; } = 100;
        [Export] public int MaxAmmo { get; set; } = 999;

        public override void _Ready()
        {
            _serverState = new ServerGameState();
        }

        /// <summary>
        /// 验证玩家移动请求
        /// </summary>
        public ValidationResult ValidateMovement(int playerId, Vector3 requestedPosition, float deltaTime)
        {
            if (!_playerStates.TryGetValue(playerId, out var playerState))
            {
                return ValidationResult.Rejected("玩家不存在");
            }

            // 计算理论最大移动距离
            float maxDistance = MaxSpeed * deltaTime;
            float actualDistance = playerState.Position.DistanceTo(requestedPosition);

            // 位置校验
            if (actualDistance > maxDistance + MaxPositionDelta)
            {
                GD.PushWarning($"玩家 {playerId} 移动距离异常: {actualDistance:F2} > {maxDistance:F2}");
                return ValidationResult.Rejected("移动距离超出限制");
            }

            // 碰撞检测（简化版）
            if (IsPositionColliding(requestedPosition))
            {
                return ValidationResult.Rejected("目标位置无效");
            }

            // 更新服务器状态
            playerState.Position = requestedPosition;
            playerState.LastMoveTime = Time.GetUnixTimeFromSystem();

            return ValidationResult.Approved();
        }

        /// <summary>
        /// 验证伤害计算
        /// </summary>
        public ValidationResult ValidateDamage(int attackerId, int targetId, int damageAmount, WeaponType weapon)
        {
            // 验证攻击者存在
            if (!_playerStates.TryGetValue(attackerId, out var attacker))
            {
                return ValidationResult.Rejected("攻击者不存在");
            }

            // 验证目标存在
            if (!_playerStates.TryGetValue(targetId, out var target))
            {
                return ValidationResult.Rejected("目标不存在");
            }

            // 验证武器伤害上限
            int maxWeaponDamage = GetMaxWeaponDamage(weapon);
            if (damageAmount > maxWeaponDamage)
            {
                GD.PushWarning($"玩家 {attackerId} 伤害异常: {damageAmount} > {maxWeaponDamage}");
                return ValidationResult.Rejected("伤害超出武器限制");
            }

            // 验证攻击距离
            float distance = attacker.Position.DistanceTo(target.Position);
            float maxRange = GetWeaponRange(weapon);
            if (distance > maxRange)
            {
                return ValidationResult.Rejected("超出攻击范围");
            }

            // 验证攻击冷却
            if (!attacker.CanAttack(weapon))
            {
                return ValidationResult.Rejected("攻击冷却中");
            }

            return ValidationResult.Approved();
        }

        /// <summary>
        /// 验证资源修改
        /// </summary>
        public ValidationResult ValidateResourceChange(int playerId, ResourceType type, int changeAmount)
        {
            if (!_playerStates.TryGetValue(playerId, out var playerState))
            {
                return ValidationResult.Rejected("玩家不存在");
            }

            // 计算修改后的值
            int currentValue = playerState.GetResource(type);
            int newValue = currentValue + changeAmount;

            // 验证下限
            if (newValue < 0)
            {
                return ValidationResult.Rejected("资源不能为负数");
            }

            // 验证上限
            int maxValue = GetMaxResource(type);
            if (newValue > maxValue)
            {
                GD.PushWarning($"玩家 {playerId} 资源 {type} 超出上限: {newValue} > {maxValue}");
                return ValidationResult.Rejected("资源超出上限");
            }

            return ValidationResult.Approved();
        }

        /// <summary>
        /// 验证状态转换
        /// </summary>
        public ValidationResult ValidateStateTransition(int playerId, PlayerState fromState, PlayerState toState)
        {
            // 定义允许的状态转换
            var allowedTransitions = new Dictionary<PlayerState, HashSet<PlayerState>>
            {
                [PlayerState.Idle] = new() { PlayerState.Moving, PlayerState.Attacking, PlayerState.Dead },
                [PlayerState.Moving] = new() { PlayerState.Idle, PlayerState.Attacking, PlayerState.Dead },
                [PlayerState.Attacking] = new() { PlayerState.Idle, PlayerState.Moving, PlayerState.Dead },
                [PlayerState.Dead] = new() { PlayerState.Idle } // 复活
            };

            if (!allowedTransitions.TryGetValue(fromState, out var allowed) || !allowed.Contains(toState))
            {
                return ValidationResult.Rejected($"非法状态转换: {fromState} -> {toState}");
            }

            return ValidationResult.Approved();
        }

        private bool IsPositionColliding(Vector3 position)
        {
            // 实现碰撞检测逻辑
            // 返回 true 如果位置无效（在墙内、在空中等）
            return false;
        }

        private int GetMaxWeaponDamage(WeaponType weapon) => weapon switch
        {
            WeaponType.Pistol => 25,
            WeaponType.Rifle => 35,
            WeaponType.Sniper => 100,
            WeaponType.Shotgun => 80,
            _ => 10
        };

        private float GetWeaponRange(WeaponType weapon) => weapon switch
        {
            WeaponType.Pistol => 50.0f,
            WeaponType.Rifle => 100.0f,
            WeaponType.Sniper => 500.0f,
            WeaponType.Shotgun => 30.0f,
            _ => 10.0f
        };

        private int GetMaxResource(ResourceType type) => type switch
        {
            ResourceType.Health => MaxHealth,
            ResourceType.Ammo => MaxAmmo,
            ResourceType.Gold => int.MaxValue,
            _ => 9999
        };
    }

    /// <summary>
    /// 验证结果
    /// </summary>
    public class ValidationResult
    {
        public bool IsValid { get; private set; }
        public string Reason { get; private set; }

        private ValidationResult(bool isValid, string reason)
        {
            IsValid = isValid;
            Reason = reason;
        }

        public static ValidationResult Approved() => new(true, string.Empty);
        public static ValidationResult Rejected(string reason) => new(false, reason);
    }

    /// <summary>
    /// 玩家服务器状态
    /// </summary>
    public class PlayerServerState
    {
        public int PlayerId { get; set; }
        public Vector3 Position { get; set; }
        public Vector3 Velocity { get; set; }
        public PlayerState CurrentState { get; set; }
        public double LastMoveTime { get; set; }
        public Dictionary<WeaponType, double> LastAttackTime { get; set; } = new();
        public Dictionary<ResourceType, int> Resources { get; set; } = new();

        public bool CanAttack(WeaponType weapon)
        {
            double now = Time.GetUnixTimeFromSystem();
            if (!LastAttackTime.TryGetValue(weapon, out double lastTime))
            {
                return true;
            }
            return (now - lastTime) >= GetAttackCooldown(weapon);
        }

        public int GetResource(ResourceType type) => Resources.GetValueOrDefault(type, 0);

        private double GetAttackCooldown(WeaponType weapon) => weapon switch
        {
            WeaponType.Pistol => 0.5,
            WeaponType.Rifle => 0.1,
            WeaponType.Sniper => 2.0,
            WeaponType.Shotgun => 1.0,
            _ => 1.0
        };
    }

    public enum PlayerState { Idle, Moving, Attacking, Dead }
    public enum WeaponType { Pistol, Rifle, Sniper, Shotgun, Melee }
    public enum ResourceType { Health, Ammo, Gold, Experience }
}

15.1.2 客户端预测验证

客户端预测需要服务器进行回滚和重放验证，确保一致性。

using Godot;
using System;
using System.Collections.Generic;
using System.Linq;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 客户端预测验证器
    /// </summary>
    public class ClientPredictionValidator
    {
        // 客户端输入历史
        private Dictionary<int, Queue<PlayerInput>> _inputHistory = new();

        // 服务器状态快照
        private Dictionary<int, StateSnapshot> _stateSnapshots = new();

        // 配置
        public int MaxInputHistory { get; set; } = 60; // 1秒 @ 60fps
        public float PositionTolerance { get; set; } = 0.5f;
        public float VelocityTolerance { get; set; } = 2.0f;

        /// <summary>
        /// 验证客户端预测状态
        /// </summary>
        public PredictionValidationResult ValidatePrediction(
            int playerId,
            StateSnapshot clientSnapshot,
            List<PlayerInput> clientInputs)
        {
            // 获取服务器对应时间点的状态
            if (!_stateSnapshots.TryGetValue(playerId, out var serverSnapshot))
            {
                return PredictionValidationResult.Failed("服务器状态不存在");
            }

            // 时间戳校验
            if (Math.Abs(serverSnapshot.Timestamp - clientSnapshot.Timestamp) > 0.1)
            {
                return PredictionValidationResult.Failed("时间戳不匹配");
            }

            // 服务器回滚并重新模拟
            var simulatedState = RollbackAndSimulate(serverSnapshot, clientInputs);

            // 比较结果
            var differences = CompareStates(simulatedState, clientSnapshot);

            if (differences.Count == 0)
            {
                return PredictionValidationResult.Success();
            }

            // 检查差异是否在可接受范围内
            bool withinTolerance = true;
            foreach (var diff in differences)
            {
                if (!IsDifferenceAcceptable(diff))
                {
                    withinTolerance = false;
                    GD.PushWarning($"玩家 {playerId} 预测差异过大: {diff.Field} = {diff.Difference:F4}");
                }
            }

            if (withinTolerance)
            {
                return PredictionValidationResult.SuccessWithCorrection(differences);
            }
            else
            {
                return PredictionValidationResult.FailedWithCorrection("预测偏差过大", simulatedState);
            }
        }

        /// <summary>
        /// 服务器回滚并重放输入
        /// </summary>
        private StateSnapshot RollbackAndSimulate(StateSnapshot baseState, List<PlayerInput> inputs)
        {
            var state = baseState.Clone();

            foreach (var input in inputs.OrderBy(i => i.SequenceNumber))
            {
                // 验证输入有效性
                if (!ValidateInput(state, input))
                {
                    GD.PushWarning($"输入 {input.SequenceNumber} 验证失败，跳过处理");
                    continue;
                }

                // 应用输入
                ApplyInput(ref state, input);

                // 物理模拟
                SimulatePhysics(ref state, input.DeltaTime);
            }

            return state;
        }

        /// <summary>
        /// 验证单个输入
        /// </summary>
        private bool ValidateInput(StateSnapshot state, PlayerInput input)
        {
            // 检查输入时间合理性
            if (input.DeltaTime <= 0 || input.DeltaTime > 0.1f)
            {
                return false;
            }

            // 检查移动输入幅度
            if (input.MoveInput.Length() > 1.0f)
            {
                return false;
            }

            // 检查视角旋转幅度
            if (Math.Abs(input.RotationDelta) > Math.PI)
            {
                return false;
            }

            return true;
        }

        /// <summary>
        /// 应用输入到状态
        /// </summary>
        private void ApplyInput(ref StateSnapshot state, PlayerInput input)
        {
            // 计算移动方向
            Vector3 moveDirection = new Vector3(input.MoveInput.X, 0, input.MoveInput.Y);
            moveDirection = moveDirection.Rotated(Vector3.Up, state.RotationY);

            // 应用加速度
            float acceleration = input.IsSprinting ? 20.0f : 10.0f;
            state.Velocity += moveDirection * acceleration * input.DeltaTime;

            // 应用旋转
            state.RotationY += input.RotationDelta;

            // 处理动作输入
            if (input.IsJumping && state.IsGrounded)
            {
                state.Velocity.Y = 10.0f;
                state.IsGrounded = false;
            }

            if (input.IsAttacking)
            {
                state.LastAttackFrame = input.SequenceNumber;
            }
        }

        /// <summary>
        /// 物理模拟
        /// </summary>
        private void SimulatePhysics(ref StateSnapshot state, float deltaTime)
        {
            // 应用重力
            if (!state.IsGrounded)
            {
                state.Velocity.Y -= 20.0f * deltaTime;
            }

            // 应用速度
            state.Position += state.Velocity * deltaTime;

            // 地面检测
            if (state.Position.Y <= 0)
            {
                state.Position = new Vector3(state.Position.X, 0, state.Position.Z);
                state.Velocity.Y = 0;
                state.IsGrounded = true;
            }

            // 速度衰减
            state.Velocity *= 0.9f;
        }

        /// <summary>
        /// 比较两个状态
        /// </summary>
        private List<StateDifference> CompareStates(StateSnapshot server, StateSnapshot client)
        {
            var differences = new List<StateDifference>();

            // 位置比较
            float positionDiff = server.Position.DistanceTo(client.Position);
            if (positionDiff > 0.01f)
            {
                differences.Add(new StateDifference("Position", positionDiff, server.Position, client.Position));
            }

            // 速度比较
            float velocityDiff = server.Velocity.DistanceTo(client.Velocity);
            if (velocityDiff > 0.01f)
            {
                differences.Add(new StateDifference("Velocity", velocityDiff, server.Velocity, client.Velocity));
            }

            // 旋转比较
            float rotationDiff = Math.Abs(server.RotationY - client.RotationY);
            rotationDiff = Math.Min(rotationDiff, (float)(2 * Math.PI - rotationDiff));
            if (rotationDiff > 0.01f)
            {
                differences.Add(new StateDifference("RotationY", rotationDiff, server.RotationY, client.RotationY));
            }

            // 状态比较
            if (server.IsGrounded != client.IsGrounded)
            {
                differences.Add(new StateDifference("IsGrounded", 1, server.IsGrounded, client.IsGrounded));
            }

            return differences;
        }

        /// <summary>
        /// 判断差异是否可接受
        /// </summary>
        private bool IsDifferenceAcceptable(StateDifference diff)
        {
            switch (diff.Field)
            {
                case "Position":
                    return diff.Difference <= PositionTolerance;
                case "Velocity":
                    return diff.Difference <= VelocityTolerance;
                case "RotationY":
                    return diff.Difference <= 0.1f; // 约5.7度
                default:
                    return diff.Difference <= 0.01f;
            }
        }

        /// <summary>
        /// 记录状态快照
        /// </summary>
        public void RecordSnapshot(int playerId, StateSnapshot snapshot)
        {
            _stateSnapshots[playerId] = snapshot;
        }

        /// <summary>
        /// 记录玩家输入
        /// </summary>
        public void RecordInput(int playerId, PlayerInput input)
        {
            if (!_inputHistory.TryGetValue(playerId, out var history))
            {
                history = new Queue<PlayerInput>();
                _inputHistory[playerId] = history;
            }

            history.Enqueue(input);

            // 保持历史长度限制
            while (history.Count > MaxInputHistory)
            {
                history.Dequeue();
            }
        }
    }

    /// <summary>
    /// 状态快照
    /// </summary>
    public class StateSnapshot
    {
        public int PlayerId { get; set; }
        public double Timestamp { get; set; }
        public uint SequenceNumber { get; set; }
        public Vector3 Position { get; set; }
        public Vector3 Velocity { get; set; }
        public float RotationY { get; set; }
        public bool IsGrounded { get; set; }
        public int LastAttackFrame { get; set; }

        public StateSnapshot Clone()
        {
            return new StateSnapshot
            {
                PlayerId = this.PlayerId,
                Timestamp = this.Timestamp,
                SequenceNumber = this.SequenceNumber,
                Position = this.Position,
                Velocity = this.Velocity,
                RotationY = this.RotationY,
                IsGrounded = this.IsGrounded,
                LastAttackFrame = this.LastAttackFrame
            };
        }
    }

    /// <summary>
    /// 玩家输入
    /// </summary>
    public class PlayerInput
    {
        public uint SequenceNumber { get; set; }
        public double Timestamp { get; set; }
        public float DeltaTime { get; set; }
        public Vector2 MoveInput { get; set; }
        public float RotationDelta { get; set; }
        public bool IsJumping { get; set; }
        public bool IsSprinting { get; set; }
        public bool IsAttacking { get; set; }
    }

    /// <summary>
    /// 状态差异
    /// </summary>
    public class StateDifference
    {
        public string Field { get; set; }
        public float Difference { get; set; }
        public object ServerValue { get; set; }
        public object ClientValue { get; set; }

        public StateDifference(string field, float difference, object serverValue, object clientValue)
        {
            Field = field;
            Difference = difference;
            ServerValue = serverValue;
            ClientValue = clientValue;
        }
    }

    /// <summary>
    /// 预测验证结果
    /// </summary>
    public class PredictionValidationResult
    {
        public bool Success { get; private set; }
        public bool RequiresCorrection { get; private set; }
        public StateSnapshot CorrectedState { get; private set; }
        public List<StateDifference> Differences { get; private set; }

15.1.3 状态同步验证

状态同步验证确保服务器和客户端之间的状态一致性。

using Godot;
using System;
using System.Collections.Generic;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 状态同步验证器
    /// </summary>
    public class StateSyncValidator : Node
    {
        // 状态校验缓存
        private Dictionary<int, SyncState> _syncStates = new();

        // 配置参数
        [Export] public float SyncInterval { get; set; } = 0.1f; // 100ms
        [Export] public int MaxSyncRetries { get; set; } = 3;
        [Export] public float DesyncThreshold { get; set; } = 2.0f;

        /// <summary>
        /// 验证同步数据包
        /// </summary>
        public SyncValidationResult ValidateSyncPacket(int playerId, SyncPacket packet)
        {
            if (!_syncStates.TryGetValue(playerId, out var syncState))
            {
                syncState = new SyncState { PlayerId = playerId };
                _syncStates[playerId] = syncState;
            }

            // 序列号验证
            if (packet.SequenceNumber <= syncState.LastReceivedSequence)
            {
                return SyncValidationResult.Rejected("重复或过期序列号");
            }

            // 时间戳验证
            double serverTime = Time.GetUnixTimeFromSystem();
            if (Math.Abs(serverTime - packet.Timestamp) > 5.0)
            {
                return SyncValidationResult.Rejected("时间戳偏差过大");
            }

            // 计算与上次状态的变化
            if (syncState.LastPosition != Vector3.Zero)
            {
                Vector3 delta = packet.Position - syncState.LastPosition;
                float timeDelta = (float)(packet.Timestamp - syncState.LastTimestamp);

                if (timeDelta > 0)
                {
                    float impliedSpeed = delta.Length() / timeDelta;

                    // 验证隐含速度
                    if (impliedSpeed > 30.0f) // 超过最大可能速度
                    {
                        GD.PushWarning($"玩家 {playerId} 状态同步速度异常: {impliedSpeed:F2}");
                        return SyncValidationResult.Rejected("移动速度异常");
                    }
                }
            }

            // 更新同步状态
            syncState.LastReceivedSequence = packet.SequenceNumber;
            syncState.LastPosition = packet.Position;
            syncState.LastTimestamp = packet.Timestamp;
            syncState.LastSyncTime = serverTime;

            return SyncValidationResult.Accepted();
        }

        /// <summary>
        /// 检查状态同步健康度
        /// </summary>
        public SyncHealth CheckSyncHealth(int playerId)
        {
            if (!_syncStates.TryGetValue(playerId, out var syncState))
            {
                return SyncHealth.Unknown;
            }

            double now = Time.GetUnixTimeFromSystem();
            double timeSinceLastSync = now - syncState.LastSyncTime;

            if (timeSinceLastSync > SyncInterval * 5)
            {
                return SyncHealth.Critical;
            }
            else if (timeSinceLastSync > SyncInterval * 2)
            {
                return SyncHealth.Degraded;
            }

            // 检查丢包率
            if (syncState.PacketsReceived > 10)
            {
                float lossRate = (float)syncState.PacketsLost / syncState.PacketsReceived;
                if (lossRate > 0.2f)
                {
                    return SyncHealth.Degraded;
                }
            }

            return SyncHealth.Healthy;
        }

        /// <summary>
        /// 生成服务器权威状态
        /// </summary>
        public ServerStatePacket GenerateAuthoritativeState(int playerId)
        {
            // 获取服务器计算的权威状态
            var serverState = CalculateServerState(playerId);

            return new ServerStatePacket
            {
                PlayerId = playerId,
                SequenceNumber = GetNextSequenceNumber(playerId),
                Timestamp = Time.GetUnixTimeFromSystem(),
                Position = serverState.Position,
                Velocity = serverState.Velocity,
                RotationY = serverState.RotationY,
                Health = serverState.Health,
                Ammo = serverState.Ammo,
                IsGrounded = serverState.IsGrounded
            };
        }

        /// <summary>
        /// 检测状态不同步
        /// </summary>
        public DesyncReport DetectDesync(int playerId, SyncPacket clientState)
        {
            var serverState = CalculateServerState(playerId);
            var report = new DesyncReport { PlayerId = playerId, Timestamp = Time.GetUnixTimeFromSystem() };

            // 位置不同步
            float positionDiff = serverState.Position.DistanceTo(clientState.Position);
            if (positionDiff > DesyncThreshold)
            {
                report.DesyncItems.Add(new DesyncItem
                {
                    Field = "Position",
                    ServerValue = serverState.Position,
                    ClientValue = clientState.Position,
                    Difference = positionDiff
                });
            }

            // 速度不同步
            float velocityDiff = serverState.Velocity.DistanceTo(clientState.Velocity);
            if (velocityDiff > 5.0f)
            {
                report.DesyncItems.Add(new DesyncItem
                {
                    Field = "Velocity",
                    ServerValue = serverState.Velocity,
                    ClientValue = clientState.Velocity,
                    Difference = velocityDiff
                });
            }

            report.IsDesynced = report.DesyncItems.Count > 0;
            return report;
        }

        private ServerState CalculateServerState(int playerId)
        {
            // 从游戏逻辑中获取服务器计算的权威状态
            // 这里简化实现
            return new ServerState
            {
                Position = Vector3.Zero,
                Velocity = Vector3.Zero,
                RotationY = 0,
                Health = 100,
                Ammo = 30,
                IsGrounded = true
            };
        }

        private uint GetNextSequenceNumber(int playerId)
        {
            if (!_syncStates.TryGetValue(playerId, out var syncState))
            {
                return 1;
            }
            return syncState.LastReceivedSequence + 1;
        }
    }

15.2 反作弊系统设计原理

15.2.0 作弊类型分类

了解作弊类型是设计反作弊系统的基础。现代游戏面临的作弊手段日益复杂，需要多层次的防护策略。

内存修改类作弊

这是最常见的作弊方式，攻击者通过内存编辑器（如Cheat Engine）修改游戏进程内存：

  • 数值修改：生命值、弹药、金币、经验等
  • 冻结数值：使生命值或弹药不减少
  • 扫描定位：通过变化值定位内存地址
  • 代码注入：注入DLL修改游戏逻辑

网络层作弊

攻击者通过拦截和篡改网络数据包实现作弊：

  • 数据包篡改：修改发送给服务器的数据
  • 重放攻击：重复发送有利的数据包
  • 延迟攻击：人为制造网络延迟获取优势
  • 中间人攻击：截获和修改通信内容

自动化作弊

通过程序模拟人类输入，获得超越人类反应能力的优势：

  • 自动瞄准（Aimbot）：自动锁定敌人头部
  • 触发器（Triggerbot）：敌人进入准星自动开火
  • 脚本宏：自动执行复杂操作序列
  • 机器人（Bot）：完全自动化游戏

信息泄露类作弊

通过修改渲染或获取服务器不应发送的信息：

  • 透视（Wallhack）：显示墙壁后的敌人
  • 雷达（Radar）：在小地图上显示所有敌人位置
  • 去雾：移除烟雾弹或视野限制效果
  • 亮度调整：强制提高暗处亮度

15.2.1 反作弊检测机制

有效的反作弊需要多层检测机制相互配合：

客户端检测

在客户端运行的检测代码，作为第一道防线：

客户端检测技术：
├── 内存完整性检查
│   └── 扫描关键数值是否异常
├── 调试器检测
│   └── 检测是否有调试器附加
├── 进程检测
│   └── 扫描已知作弊软件进程
├── 代码完整性
│   └── 校验关键代码段未被修改
└── 行为分析
    └── 检测异常输入模式

注意：客户端检测只能作为辅助手段，不可信任。所有关键检测必须在服务器进行。

服务器端验证

服务器端验证是最可靠的检测方式：

  • 物理验证：检查移动距离、速度是否超出物理限制
  • 状态验证：检查游戏状态转换是否合法
  • 逻辑验证：检查伤害计算、掉落概率是否正确
  • 统计分析：分析玩家行为数据，识别异常模式

行为分析系统

通过机器学习或统计方法识别异常行为：

15.2.2 封禁策略

发现作弊后的处理策略：

临时封禁

  • 短期封禁：24小时，用于轻度违规或疑似作弊
  • 冷却期：给玩家改过自新的机会
  • 观察期：解封后重点监控该玩家

永久封禁

  • 账号封禁：封禁游戏账号
  • 硬件封禁：封禁设备指纹或硬件ID
  • IP封禁：封禁IP地址（注意：可能误伤公共网络）

渐进式惩罚

违规次数 → 惩罚措施
第1次：警告 + 短期封禁（1天）
第2次：中期封禁（7天）
第3次：长期封禁（30天）
第4次：永久封禁

15.2.3 隐私与安全的平衡

反作弊措施可能涉及隐私问题，需要谨慎处理：

数据收集边界

  • 可收集：游戏内行为数据、游戏环境信息
  • 敏感：个人文件、浏览历史、其他应用数据
  • 禁止：密码、支付信息、个人通讯内容

透明度原则

  • 在隐私政策中明确说明反作弊数据收集
  • 告知用户何种行为会触发封禁
  • 提供申诉渠道

15.2.4 性能与权衡

反作弊系统本身也会带来性能开销：

验证计算开销

  • 位置验证：O(1) 简单距离计算
  • 碰撞检测：O(n) 视场景复杂度
  • 统计分析：O(n) 需要历史数据支持

优化策略

  • 分层验证：简单检查先执行，复杂检查异步执行
  • 采样验证：并非每次操作都全量验证
  • 增量更新：只验证变化的部分

网络延迟与安全严格度

严格的服务器验证会增加延迟：

15.3 包序列号与防重放

序列号机制是防止重放攻击和数据包篡改的基础安全手段。

15.2.1 序列号机制

序列号确保数据包按序到达且不被重复处理。

using Godot;
using System;
using System.Collections.Generic;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 数据包序列号管理器
    /// </summary>
    public class SequenceNumberManager
    {
        // 每个连接的序列号状态
        private Dictionary<int, SequenceState> _sequenceStates = new();

        // 配置
        public uint WindowSize { get; set; } = 32; // 滑动窗口大小
        public uint MaxSequenceNumber { get; set; } = uint.MaxValue;

        /// <summary>
        /// 初始化连接序列号
        /// </summary>
        public void InitializeConnection(int connectionId)
        {
            _sequenceStates[connectionId] = new SequenceState
            {
                ConnectionId = connectionId,
                ExpectedSequence = 1,
                HighestReceived = 0,
                ReceivedWindow = new bool[WindowSize]
            };
        }

        /// <summary>
        /// 验证数据包序列号
        /// </summary>
        public SequenceValidationResult ValidateSequence(int connectionId, uint sequenceNumber)
        {
            if (!_sequenceStates.TryGetValue(connectionId, out var state))
            {
                return SequenceValidationResult.Rejected("连接未初始化");
            }

            // 检查序列号是否已回绕
            if (sequenceNumber < state.ExpectedSequence &&
                state.ExpectedSequence - sequenceNumber > MaxSequenceNumber / 2)
            {
                return SequenceValidationResult.Rejected("序列号回绕异常");
            }

            // 检查是否在窗口范围内（允许乱序）
            if (IsWithinWindow(state, sequenceNumber))
            {
                int windowIndex = GetWindowIndex(state, sequenceNumber);

                // 检查是否已接收
                if (state.ReceivedWindow[windowIndex])
                {
                    return SequenceValidationResult.Rejected("重复数据包");
                }

                // 标记为已接收
                state.ReceivedWindow[windowIndex] = true;

                // 更新最高接收序列号
                if (sequenceNumber > state.HighestReceived)
                {
                    state.HighestReceived = sequenceNumber;
                }

                // 处理按序到达的数据包
                ProcessInOrderPackets(state);

                return SequenceValidationResult.Accepted();
            }
            else if (sequenceNumber < state.ExpectedSequence)
            {
                // 序列号过小，可能是重放攻击
                return SequenceValidationResult.Rejected("序列号过小（可能的重放攻击）");
            }
            else
            {
                // 序列号过大，窗口外
                return SequenceValidationResult.Rejected("序列号超出窗口");
            }
        }

        /// <summary>
        /// 获取下一个发送序列号
        /// </summary>
        public uint GetNextSequenceNumber(int connectionId)
        {
            if (!_sequenceStates.TryGetValue(connectionId, out var state))
            {
                InitializeConnection(connectionId);
                state = _sequenceStates[connectionId];
            }

            state.NextSendSequence++;
            return state.NextSendSequence;
        }

        /// <summary>
        /// 获取期望的接收序列号
        /// </summary>
        public uint GetExpectedSequence(int connectionId)
        {
            if (_sequenceStates.TryGetValue(connectionId, out var state))
            {
                return state.ExpectedSequence;
            }
            return 1;
        }

        /// <summary>
        /// 检查序列号是否在窗口内
        /// </summary>
        private bool IsWithinWindow(SequenceState state, uint sequenceNumber)
        {
            if (sequenceNumber >= state.ExpectedSequence)
            {
                return sequenceNumber - state.ExpectedSequence < WindowSize;
            }
            else
            {
                // 处理回绕情况
                return state.ExpectedSequence - sequenceNumber > MaxSequenceNumber / 2;
            }
        }

        /// <summary>
        /// 获取窗口索引
        /// </summary>
        private int GetWindowIndex(SequenceState state, uint sequenceNumber)
        {
            return (int)(sequenceNumber % WindowSize);
        }

        /// <summary>
        /// 处理按序到达的数据包
        /// </summary>
        private void ProcessInOrderPackets(SequenceState state)
        {
            while (true)
            {
                int index = GetWindowIndex(state, state.ExpectedSequence);
                if (!state.ReceivedWindow[index])
                {
                    break;
                }

                // 清除标记
                state.ReceivedWindow[index] = false;
                state.ExpectedSequence++;

                if (state.ExpectedSequence == 0)
                {
                    // 处理回绕
                    state.ExpectedSequence = 1;
                }
            }
        }

        /// <summary>
        /// 清理断开的连接
        /// </summary>
        public void RemoveConnection(int connectionId)
        {
            _sequenceStates.Remove(connectionId);
        }

        /// <summary>
        /// 获取连接统计信息
        /// </summary>
        public SequenceStatistics GetStatistics(int connectionId)
        {
            if (!_sequenceStates.TryGetValue(connectionId, out var state))
            {
                return null;
            }

            return new SequenceStatistics
            {
                ConnectionId = connectionId,
                ExpectedSequence = state.ExpectedSequence,
                HighestReceived = state.HighestReceived,
                PacketsAccepted = state.PacketsAccepted,
                PacketsRejected = state.PacketsRejected,
                OutOfOrderCount = state.OutOfOrderCount
            };
        }
    }

    /// <summary>
    /// 序列号状态
    /// </summary>
    public class SequenceState
    {
        public int ConnectionId { get; set; }
        public uint ExpectedSequence { get; set; } = 1; // 期望接收的下一个序列号
        public uint HighestReceived { get; set; } = 0;  // 收到的最高序列号
        public uint NextSendSequence { get; set; } = 0; // 下一个发送序列号
        public bool[] ReceivedWindow { get; set; }

        // 统计信息
        public ulong PacketsAccepted { get; set; }
        public ulong PacketsRejected { get; set; }
        public ulong OutOfOrderCount { get; set; }
    }

    /// <summary>
    /// 序列号统计
    /// </summary>
    public class SequenceStatistics
    {
        public int ConnectionId { get; set; }
        public uint ExpectedSequence { get; set; }
        public uint HighestReceived { get; set; }
        public ulong PacketsAccepted { get; set; }
        public ulong PacketsRejected { get; set; }
        public ulong OutOfOrderCount { get; set; }

        public float RejectionRate => PacketsAccepted + PacketsRejected > 0
            ? (float)PacketsRejected / (PacketsAccepted + PacketsRejected)
            : 0;
    }

    /// <summary>
    /// 序列号验证结果
    /// </summary>
    public class SequenceValidationResult
    {
        public bool IsValid { get; private set; }

15.2.2 时间戳验证

时间戳防止延迟攻击和过期数据包处理。

using Godot;
using System;
using System.Collections.Generic;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 时间戳验证管理器
    /// </summary>
    public class TimestampValidator
    {
        // 连接时间戳状态
        private Dictionary<int, TimestampState> _timestampStates = new();

        // 配置
        public double MaxClockSkew { get; set; } = 5.0; // 最大时钟偏差（秒）
        public double PacketTimeout { get; set; } = 10.0; // 数据包超时（秒）
        public double MinPacketInterval { get; set; } = 0.001; // 最小包间隔（秒）

        /// <summary>
        /// 初始化连接时间戳
        /// </summary>
        public void InitializeConnection(int connectionId)
        {
            _timestampStates[connectionId] = new TimestampState
            {
                ConnectionId = connectionId,
                ClockOffset = 0,
                LastReceivedTime = 0,
                IsClockSynced = false
            };
        }

        /// <summary>
        /// 同步时钟（使用NTP风格算法）
        /// </summary>
        public void SynchronizeClock(int connectionId, double clientSendTime, double serverReceiveTime)
        {
            if (!_timestampStates.TryGetValue(connectionId, out var state))
            {
                InitializeConnection(connectionId);
                state = _timestampStates[connectionId];
            }

            // 计算单程延迟（简化，假设对称延迟）
            double rtt = serverReceiveTime - clientSendTime;
            double oneWayDelay = rtt / 2.0;

            // 计算时钟偏移
            double estimatedServerTime = clientSendTime + oneWayDelay;
            state.ClockOffset = serverReceiveTime - estimatedServerTime;
            state.IsClockSynced = true;
            state.LastSyncTime = serverReceiveTime;

            GD.Print($"连接 {connectionId} 时钟同步: 偏移={state.ClockOffset:F6}秒, RTT={rtt:F6}秒");
        }

        /// <summary>
        /// 验证数据包时间戳
        /// </summary>
        public TimestampValidationResult ValidateTimestamp(int connectionId, double packetTimestamp)
        {
            if (!_timestampStates.TryGetValue(connectionId, out var state))
            {
                return TimestampValidationResult.Rejected("连接未初始化");
            }

            double serverTime = Time.GetUnixTimeFromSystem();

            // 未同步时钟时的宽松验证
            if (!state.IsClockSynced)
            {
                if (packetTimestamp > serverTime + MaxClockSkew)
                {
                    return TimestampValidationResult.Rejected("时间戳超前过多（未同步）");
                }
                return TimestampValidationResult.Accepted();
            }

            // 校正客户端时间戳
            double adjustedTimestamp = packetTimestamp + state.ClockOffset;
            double timeDiff = serverTime - adjustedTimestamp;

            // 检查时间戳是否在未来
            if (timeDiff < -MaxClockSkew)
            {
                return TimestampValidationResult.Rejected($"时间戳超前: {-timeDiff:F3}秒");
            }

            // 检查数据包是否过期
            if (timeDiff > PacketTimeout)
            {
                return TimestampValidationResult.Rejected($"数据包过期: {timeDiff:F3}秒");
            }

            // 检查包间隔（防止洪水攻击）
            if (state.LastReceivedTime > 0)
            {
                double interval = adjustedTimestamp - state.LastReceivedTime;
                if (interval < MinPacketInterval && interval > 0)
                {
                    state.FastPacketCount++;
                    if (state.FastPacketCount > 10)
                    {
                        return TimestampValidationResult.Rejected("包发送频率异常");
                    }
                }
                else
                {
                    state.FastPacketCount = 0;
                }
            }

            // 更新时间戳
            if (adjustedTimestamp > state.LastReceivedTime)
            {
                state.LastReceivedTime = adjustedTimestamp;
            }

            state.PacketsAccepted++;
            return TimestampValidationResult.Accepted();
        }

        /// <summary>
        /// 验证时间戳和序列号组合
        /// </summary>
        public CombinedValidationResult ValidateTimestampAndSequence(
            int connectionId,
            double packetTimestamp,
            uint sequenceNumber)
        {
            var timestampResult = ValidateTimestamp(connectionId, packetTimestamp);
            if (!timestampResult.IsValid)
            {
                return CombinedValidationResult.Rejected(timestampResult.Reason);
            }

            // 验证序列号（需要与SequenceNumberManager配合使用）
            // 这里简化处理

            return CombinedValidationResult.Accepted();
        }

        /// <summary>
        /// 检查是否需要重新同步时钟
        /// </summary>
        public bool NeedsResync(int connectionId)
        {
            if (!_timestampStates.TryGetValue(connectionId, out var state))
            {
                return true;
            }

            if (!state.IsClockSynced)
            {
                return true;
            }

            // 每隔一段时间重新同步
            double serverTime = Time.GetUnixTimeFromSystem();
            return (serverTime - state.LastSyncTime) > 60.0; // 每分钟
        }

        /// <summary>
        /// 获取连接时间戳状态
        /// </summary>
        public TimestampState GetTimestampState(int connectionId)
        {
            return _timestampStates.GetValueOrDefault(connectionId);
        }

        /// <summary>
        /// 计算数据包延迟
        /// </summary>
        public double CalculateLatency(int connectionId, double packetTimestamp, double receiveTime)
        {
            if (!_timestampStates.TryGetValue(connectionId, out var state) || !state.IsClockSynced)
            {
                return -1; // 未知
            }

            double adjustedTimestamp = packetTimestamp + state.ClockOffset;
            return receiveTime - adjustedTimestamp;
        }
    }

    /// </summary>

        public int ConnectionId { get; set; }
        public double ClockOffset { get; set; } // 客户端时钟偏移（秒）

15.2.3 重放攻击防护

重放攻击防护结合序列号和时间戳，确保数据包的独特性。

using Godot;
using System;
using System.Collections.Generic;
using System.Security.Cryptography;
using System.Text;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 防重放攻击管理器
    /// </summary>
    public class ReplayProtectionManager
    {
        // 每个连接的防重放状态
        private Dictionary<int, ReplayState> _replayStates = new();

        // 全局唯一包ID缓存
        private HashSet<string> _globalPacketIds = new();

        // 配置
        public TimeSpan PacketIdRetentionTime { get; set; } = TimeSpan.FromMinutes(5);
        public int MaxStoredPacketIds { get; set; } = 10000;

        /// <summary>
        /// 初始化连接
        /// </summary>
        public void InitializeConnection(int connectionId)
        {
            _replayStates[connectionId] = new ReplayState
            {
                ConnectionId = connectionId,
                ReceivedPacketIds = new HashSet<string>(),
                PacketHistory = new Queue<PacketHistoryEntry>()
            };
        }

        /// <summary>
        /// 生成唯一包标识
        /// </summary>
        public string GeneratePacketId(int connectionId, uint sequenceNumber, double timestamp, byte[] payload)
        {
            // 组合多个因素生成唯一ID
            var data = new StringBuilder();
            data.Append(connectionId);
            data.Append(sequenceNumber);
            data.Append(timestamp.ToString("F6"));

            if (payload != null)
            {
                data.Append(Convert.ToHexString(payload));
            }

            // 使用HMAC增加安全性
            using (var hmac = new HMACSHA256(Encoding.UTF8.GetBytes("your-secret-key")))
            {
                byte[] hash = hmac.ComputeHash(Encoding.UTF8.GetBytes(data.ToString()));
                return Convert.ToBase64String(hash).Substring(0, 24); // 取前24个字符
            }
        }

        /// <summary>
        /// 验证数据包是否为新包
        /// </summary>
        public ReplayValidationResult ValidatePacket(int connectionId, string packetId)
        {
            if (!_replayStates.TryGetValue(connectionId, out var state))
            {
                InitializeConnection(connectionId);
                state = _replayStates[connectionId];
            }

            // 检查是否已接收过
            if (state.ReceivedPacketIds.Contains(packetId))
            {
                state.ReplayAttempts++;
                return ReplayValidationResult.Rejected("检测到重放数据包", isReplay: true);
            }

            // 检查全局缓存
            if (_globalPacketIds.Contains(packetId))
            {
                state.ReplayAttempts++;
                return ReplayValidationResult.Rejected("全局检测重放数据包", isReplay: true);
            }

            // 添加到新记录
            state.ReceivedPacketIds.Add(packetId);
            _globalPacketIds.Add(packetId);

            // 记录历史
            state.PacketHistory.Enqueue(new PacketHistoryEntry
            {
                PacketId = packetId,
                Timestamp = Time.GetUnixTimeFromSystem()
            });

            // 清理过期记录
            CleanupOldEntries(state);
            CleanupGlobalCache();

            state.PacketsAccepted++;
            return ReplayValidationResult.Accepted();
        }

        /// <summary>
        /// 完整验证流程
        /// </summary>
        public ReplayValidationResult FullValidation(
            int connectionId,
            uint sequenceNumber,
            double timestamp,
            byte[] payload)
        {
            // 生成包ID
            string packetId = GeneratePacketId(connectionId, sequenceNumber, timestamp, payload);

            // 验证重放
            return ValidatePacket(connectionId, packetId);
        }

        /// <summary>
        /// 批量验证（用于处理乱序到达的数据包）
        /// </summary>
        public List<ReplayValidationResult> ValidateBatch(int connectionId, List<PacketInfo> packets)
        {
            var results = new List<ReplayValidationResult>();
            var acceptedPackets = new List<PacketInfo>();

            // 第一遍：检查重放
            foreach (var packet in packets)
            {
                string packetId = GeneratePacketId(connectionId, packet.SequenceNumber, packet.Timestamp, packet.Payload);
                var result = ValidatePacket(connectionId, packetId);
                results.Add(result);

                if (result.IsValid)
                {
                    acceptedPackets.Add(packet);
                }
            }

            // 第二遍：检查序列号逻辑（可选）
            // 这里可以实现更复杂的序列号验证逻辑

            return results;
        }

        /// <summary>
        /// 清理过期的历史记录
        /// </summary>
        private void CleanupOldEntries(ReplayState state)
        {
            double now = Time.GetUnixTimeFromSystem();

            while (state.PacketHistory.Count > 0)
            {
                var oldest = state.PacketHistory.Peek();
                if (now - oldest.Timestamp > PacketIdRetentionTime.TotalSeconds)
                {
                    state.PacketHistory.Dequeue();
                    state.ReceivedPacketIds.Remove(oldest.PacketId);
                }
                else
                {
                    break;
                }
            }

            // 限制最大数量
            while (state.ReceivedPacketIds.Count > MaxStoredPacketIds / 2)
            {
                if (state.PacketHistory.Count > 0)
                {
                    var old = state.PacketHistory.Dequeue();
                    state.ReceivedPacketIds.Remove(old.PacketId);
                }
                else
                {
                    break;
                }
            }
        }

        /// <summary>
        /// 清理全局缓存
        /// </summary>
        private void CleanupGlobalCache()
        {
            // 简化实现：当超过限制时清空
            if (_globalPacketIds.Count > MaxStoredPacketIds)
            {
                _globalPacketIds.Clear();
                GD.Print("清理全局包ID缓存");
            }
        }

        /// <summary>
        /// 获取防重放统计
        /// </summary>
        public ReplayStatistics GetStatistics(int connectionId)
        {
            if (!_replayStates.TryGetValue(connectionId, out var state))
            {
                return null;
            }

            return new ReplayStatistics
            {
                ConnectionId = connectionId,
                PacketsAccepted = state.PacketsAccepted,
                ReplayAttempts = state.ReplayAttempts,
                UniquePacketIds = state.ReceivedPacketIds.Count,
                HistorySize = state.PacketHistory.Count
            };
        }

        /// <summary>
        /// 重置连接状态
        /// </summary>
        public void ResetConnection(int connectionId)
        {
            if (_replayStates.TryGetValue(connectionId, out var state))
            {
                // 从全局缓存中移除该连接的包ID
                foreach (var entry in state.PacketHistory)
                {
                    _globalPacketIds.Remove(entry.PacketId);
                }

                _replayStates.Remove(connectionId);
            }
        }
    }

    /// <summary>
    /// 防重放状态
    /// </summary>
    public class ReplayState
    {
        public int ConnectionId { get; set; }
        public HashSet<string> ReceivedPacketIds { get; set; }
        public Queue<PacketHistoryEntry> PacketHistory { get; set; }
        public ulong PacketsAccepted { get; set; }
        public ulong ReplayAttempts { get; set; }
    }

    /// <summary>
    /// 数据包历史记录
    /// </summary>
    public class PacketHistoryEntry
    {
        public string PacketId { get; set; }
        public double Timestamp { get; set; }
    }

    /// <summary>
    /// 数据包信息
    /// </summary>
    public class PacketInfo
    {
        public uint SequenceNumber { get; set; }
        public double Timestamp { get; set; }
        public byte[] Payload { get; set; }
    }

    /// <summary>
    /// 防重放统计
    /// </summary>
    public class ReplayStatistics
    {
        public int ConnectionId { get; set; }
        public ulong PacketsAccepted { get; set; }
        public ulong ReplayAttempts { get; set; }
        public int UniquePacketIds { get; set; }
        public int HistorySize { get; set; }

        public float ReplayRate => PacketsAccepted + ReplayAttempts > 0
            ? (float)ReplayAttempts / (PacketsAccepted + ReplayAttempts)
            : 0;
    }

    /// <summary>
    /// 重放验证结果
    /// </summary>
    public class ReplayValidationResult
    {
        public bool IsValid { get; private set; }
        public string Reason { get; private set; }
        public bool IsReplay { get; private set; }
        public string PacketId { get; set; }

        private ReplayValidationResult(bool isValid, string reason, bool isReplay = false)
        {
            IsValid = isValid;
            Reason = reason;
            IsReplay = isReplay;
        }

        public static ReplayValidationResult Accepted() => new(true);
        public static ReplayValidationResult Rejected(string reason, bool isReplay = false) =>
            new(false, reason, isReplay);
    }

15.3 基础反作弊策略

反作弊系统是维护游戏公平性的最后一道防线。

15.3.1 速度检测

速度检测识别异常的移动速度。

using Godot;
using System;
using System.Collections.Generic;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 速度检测系统
    /// </summary>
    public class SpeedDetectionSystem : Node
    {
        // 玩家移动历史
        private Dictionary<int, MovementHistory> _movementHistories = new();

        // 配置参数
        [Export] public float MaxWalkSpeed { get; set; } = 5.0f;
        [Export] public float MaxSprintSpeed { get; set; } = 10.0f;
        [Export] public float MaxVehicleSpeed { get; set; } = 30.0f;
        [Export] public float SpeedTolerance { get; set; } = 1.0f; // 速度容差
        [Export] public int HistorySize { get; set; } = 60;
        [Export] public float TeleportThreshold { get; set; } = 50.0f;

        /// <summary>
        /// 记录玩家位置
        /// </summary>
        public void RecordPosition(int playerId, Vector3 position, MovementType movementType)
        {
            if (!_movementHistories.TryGetValue(playerId, out var history))
            {
                history = new MovementHistory { PlayerId = playerId };
                _movementHistories[playerId] = history;
            }

            var entry = new MovementEntry
            {
                Timestamp = Time.GetUnixTimeFromSystem(),
                Position = position,
                MovementType = movementType
            };

            history.Entries.Enqueue(entry);

            // 保持历史大小
            while (history.Entries.Count > HistorySize)
            {
                history.Entries.Dequeue();
            }

            // 分析速度
            AnalyzeSpeed(playerId, history);
        }

        /// <summary>
        /// 分析玩家速度
        /// </summary>
        private void AnalyzeSpeed(int playerId, MovementHistory history)
        {
            if (history.Entries.Count < 2)
            {
                return;
            }

            var entries = new List<MovementEntry>(history.Entries);

            // 计算瞬时速度
            for (int i = 1; i < entries.Count; i++)
            {
                var current = entries[i];
                var previous = entries[i - 1];

                float distance = current.Position.DistanceTo(previous.Position);
                double timeDelta = current.Timestamp - previous.Timestamp;

                if (timeDelta <= 0) continue;

                float instantSpeed = (float)(distance / timeDelta);

                // 检测瞬移
                if (distance > TeleportThreshold && timeDelta < 1.0)
                {
                    OnTeleportDetected(playerId, previous.Position, current.Position, distance);
                }

                // 检测超速
                float maxAllowedSpeed = GetMaxAllowedSpeed(current.MovementType);
                if (instantSpeed > maxAllowedSpeed + SpeedTolerance)
                {
                    OnSpeedViolation(playerId, instantSpeed, maxAllowedSpeed, current.MovementType);
                }

                // 更新平均速度
                history.AverageSpeed = (history.AverageSpeed * (i - 1) + instantSpeed) / i;
            }

            // 检测持续超速
            if (entries.Count >= 10)
            {
                CheckSustainedSpeedViolation(playerId, history);
            }
        }

        /// <summary>
        /// 获取最大允许速度
        /// </summary>
        private float GetMaxAllowedSpeed(MovementType type) => type switch
        {
            MovementType.Walking => MaxWalkSpeed,
            MovementType.Sprinting => MaxSprintSpeed,
            MovementType.Vehicle => MaxVehicleSpeed,
            MovementType.Flying => MaxSprintSpeed * 2, // 飞行模式
            _ => MaxWalkSpeed
        };

        /// <summary>
        /// 检测持续超速违规
        /// </summary>
        private void CheckSustainedSpeedViolation(int playerId, MovementHistory history)
        {
            var entries = new List<MovementEntry>(history.Entries);
            int violations = 0;
            float violationRatio = 0;

            for (int i = 1; i < entries.Count; i++)
            {
                var current = entries[i];
                var previous = entries[i - 1];

                float distance = current.Position.DistanceTo(previous.Position);
                double timeDelta = current.Timestamp - previous.Timestamp;

                if (timeDelta > 0)
                {
                    float speed = (float)(distance / timeDelta);
                    float maxSpeed = GetMaxAllowedSpeed(current.MovementType);

                    if (speed > maxSpeed + SpeedTolerance)
                    {
                        violations++;
                    }
                }
            }

            violationRatio = (float)violations / (entries.Count - 1);

            // 如果90%以上的记录都超速，标记为作弊
            if (violationRatio > 0.9f)
            {
                OnSustainedSpeedCheatDetected(playerId, violationRatio);
            }

            // 更新可疑度评分
            history.SuspicionScore = Math.Min(100, history.SuspicionScore + (int)(violationRatio * 10));
        }

        /// <summary>
        /// 处理瞬移检测
        /// </summary>
        private void OnTeleportDetected(int playerId, Vector3 from, Vector3 to, float distance)
        {
            var violation = new SpeedViolation
            {
                PlayerId = playerId,
                Type = ViolationType.Teleport,
                Timestamp = Time.GetUnixTimeFromSystem(),
                Details = $"瞬移: {from} -> {to}, 距离: {distance:F2}m"
            };

            GD.PushWarning($"检测到玩家 {playerId} 瞬移行为: {distance:F2}m");
            EmitSignal(SignalName.TeleportDetected, playerId, from, to, distance);
        }

        /// <summary>
        /// 处理速度违规
        /// </summary>
        private void OnSpeedViolation(int playerId, float actualSpeed, float maxSpeed, MovementType type)
        {
            var violation = new SpeedViolation
            {
                PlayerId = playerId,
                Type = ViolationType.SpeedHack,
                Timestamp = Time.GetUnixTimeFromSystem(),
                ActualSpeed = actualSpeed,
                ExpectedSpeed = maxSpeed,
                Details = $"超速: {actualSpeed:F2}m/s > {maxSpeed:F2}m/s ({type})"
            };

            GD.PushWarning($"检测到玩家 {playerId} 超速: {actualSpeed:F2} > {maxSpeed:F2} ({type})");
            EmitSignal(SignalName.SpeedViolationDetected, playerId, actualSpeed, maxSpeed);
        }

        /// <summary>
        /// 处理持续速度作弊
        /// </summary>
        private void OnSustainedSpeedCheatDetected(int playerId, float violationRatio)
        {
            GD.PushError($"玩家 {playerId} 疑似速度作弊，违规率: {violationRatio:P}");
            EmitSignal(SignalName.SustainedSpeedCheat, playerId, violationRatio);
        }

        /// <summary>
        /// 获取玩家可疑度评分
        /// </summary>
        public int GetSuspicionScore(int playerId)
        {
            if (_movementHistories.TryGetValue(playerId, out var history))
            {
                return history.SuspicionScore;
            }
            return 0;
        }

        /// <summary>
        /// 重置玩家数据
        /// </summary>
        public void ResetPlayer(int playerId)
        {
            _movementHistories.Remove(playerId);
        }

        [Signal]
        public delegate void TeleportDetectedEventHandler(int playerId, Vector3 from, Vector3 to, float distance);

        [Signal]
        public delegate void SpeedViolationDetectedEventHandler(int playerId, float actualSpeed, float maxSpeed);

        [Signal]
        public delegate void SustainedSpeedCheatEventHandler(int playerId, float violationRatio);
    }

        [Export] public float MaxJumpHeight { get; set; } = 3.0f;
        [Export] public float CollisionRadius { get; set; } = 0.5f;

        /// <summary>
        /// 验证位置有效性
        /// </summary>
        public PositionValidationResult ValidatePosition(Vector3 position, Vector3? previousPosition = null)
        {
            // 1. 检查世界边界
            if (!IsWithinWorldBounds(position))
            {
                return PositionValidationResult.Invalid("超出世界边界");
            }

            // 2. 检查禁区
            foreach (var zone in _restrictedZones)
            {
                if (zone.Contains(position))
                {
                    if (!zone.IsValidEntry(position))
                    {
                        return PositionValidationResult.Invalid($"位于禁区: {zone.Name}");
                    }
                }
            }

            // 3. 检查是否在地面
            if (!IsGrounded(position))
            {
                // 如果前一位置存在，检查是否合法跳跃
                if (previousPosition.HasValue)
                {
                    if (!IsValidJump(previousPosition.Value, position))
                    {
                        return PositionValidationResult.Invalid("飞行/悬浮异常");
                    }
                }
                else
                {
                    return PositionValidationResult.Invalid("位置不在地面上");
                }
            }

            // 4. 检查是否穿墙
            if (previousPosition.HasValue)
            {
                if (IsPassingThroughWall(previousPosition.Value, position))
                {
                    return PositionValidationResult.Invalid("穿墙检测");
                }
            }

            // 5. 检查导航网格（如果有）
            if (_navigationMesh != null)
            {
                if (!IsOnNavMesh(position))
                {
                    return PositionValidationResult.Invalid("不在可行走区域");
                }
            }

            return PositionValidationResult.Valid();
        }

        /// <summary>
        /// 检查是否在世界边界内
        /// </summary>
        private bool IsWithinWorldBounds(Vector3 position)
        {
            return position.X >= WorldMinBounds.X && position.X <= WorldMaxBounds.X &&
                   position.Y >= WorldMinBounds.Y && position.Y <= WorldMaxBounds.Y &&
                   position.Z >= WorldMinBounds.Z && position.Z <= WorldMaxBounds.Z;
        }

        /// <summary>
        /// 检查是否在地面上
        /// </summary>
        private bool IsGrounded(Vector3 position)
        {
            // 简化实现：检查Y坐标是否在地面附近
            // 实际应该使用物理射线检测
            return Math.Abs(position.Y - FloorHeight) < 0.1f ||
                   IsOnSurface(position);
        }

        /// <summary>
        /// 检查是否在表面上
        /// </summary>
        private bool IsOnSurface(Vector3 position)
        {
            // 模拟射线检测
            // 实际应使用 PhysicsServer.SpaceIntersectsRay
            return position.Y >= FloorHeight && position.Y <= FloorHeight + 0.1f;
        }

        /// <summary>
        /// 检查是否是合法跳跃
        /// </summary>
        private bool IsValidJump(Vector3 from, Vector3 to)
        {
            float heightDiff = to.Y - from.Y;

            // 高度差在跳跃范围内
            if (heightDiff < 0 || heightDiff > MaxJumpHeight)
            {
                return false;
            }

            // 水平距离合理
            Vector2 from2D = new(from.X, from.Z);
            Vector2 to2D = new(to.X, to.Z);
            float horizontalDist = from2D.DistanceTo(to2D);

            // 假设最大水平跳跃距离约为高度的2倍
            if (horizontalDist > heightDiff * 2)
            {
                return false;
            }

            return true;
        }

        /// <summary>
        /// 纠正无效位置
        /// </summary>
        public Vector3 CorrectPosition(Vector3 invalidPosition, Vector3 lastValidPosition)
        {
            // 尝试找到最近的合法位置

            // 1. 首先尝试回到上一个合法位置
            if (ValidatePosition(lastValidPosition).IsValid)
            {
                return lastValidPosition;
            }

            // 2. 向下查找地面
            Vector3 grounded = new(invalidPosition.X, FloorHeight, invalidPosition.Z);
            if (ValidatePosition(grounded).IsValid)
            {
                return grounded;
            }

            // 3. 向外扩散查找
            for (float radius = 1.0f; radius <= 10.0f; radius += 1.0f)
            {
                for (int angle = 0; angle < 360; angle += 45)
                {
                    float rad = Mathf.DegToRad(angle);
                    Vector3 candidate = new(
                        invalidPosition.X + Mathf.Cos(rad) * radius,
                        invalidPosition.Y,
                        invalidPosition.Z + Mathf.Sin(rad) * radius
                    );

                    if (ValidatePosition(candidate).IsValid)
                    {
                        return candidate;
                    }
                }
            }

            // 4. 返回原点
            return Vector3.Zero;
        }

        }

15.3.3 异常行为检测

异常行为检测通过机器学习或规则系统识别可疑模式。

using Godot;
using System;
using System.Collections.Generic;
using System.Linq;

namespace GameFramework.Network.Security
{
    /// <summary>
    /// 异常行为检测系统
    /// </summary>
    public class AnomalyDetectionSystem : Node
    {
        // 玩家行为档案
        private Dictionary<int, PlayerBehaviorProfile> _behaviorProfiles = new();

        // 检测配置
        [Export] public int ProfileWindowSize { get; set; } = 100;
        [Export] public float AnomalyThreshold { get; set; } = 0.8f;
        [Export] public int ConsecutiveAnomalyThreshold { get; set; } = 5;

        // 事件时间窗口（秒）
        [Export] public double EventWindow { get; set; } = 60.0;

        /// <summary>
        /// 记录玩家事件
        /// </summary>
        public void RecordEvent(int playerId, GameEvent gameEvent)
        {
            if (!_behaviorProfiles.TryGetValue(playerId, out var profile))
            {
                profile = new PlayerBehaviorProfile { PlayerId = playerId };
                _behaviorProfiles[playerId] = profile;
            }

            // 添加事件
            profile.RecentEvents.Enqueue(gameEvent);

            // 清理过期事件
            double now = Time.GetUnixTimeFromSystem();
            while (profile.RecentEvents.Count > 0 &&
                   now - profile.RecentEvents.Peek().Timestamp > EventWindow)
            {
                profile.RecentEvents.Dequeue();
            }

            // 更新统计数据
            UpdateStatistics(profile, gameEvent);

            // 检测异常
            DetectAnomalies(playerId, profile);
        }

        /// <summary>
        /// 更新行为统计
        /// </summary>
        private void UpdateStatistics(PlayerBehaviorProfile profile, GameEvent gameEvent)
        {
            switch (gameEvent.Type)
            {
                case EventType.DamageDealt:
                    profile.TotalDamage += (int)gameEvent.Value;
                    profile.HitCount++;
                    if (gameEvent.Value > profile.MaxDamage)
                    {
                        profile.MaxDamage = (int)gameEvent.Value;
                    }
                    break;

                case EventType.Kill:
                    profile.KillCount++;
                    break;

                case EventType.Death:
                    profile.DeathCount++;
                    break;

                case EventType.Headshot:
                    profile.HeadshotCount++;
                    break;

                case EventType.Accuracy:
                    profile.AccuracySum += gameEvent.Value;
                    profile.AccuracyCount++;
                    break;
            }

            // 计算比率
            if (profile.HitCount > 0)
            {
                profile.AverageAccuracy = profile.AccuracySum / profile.AccuracyCount;
                profile.HeadshotRatio = (float)profile.HeadshotCount / profile.HitCount;
            }

            if (profile.DeathCount > 0)
            {
                profile.KDRatio = (float)profile.KillCount / profile.DeathCount;
            }
            else
            {
                profile.KDRatio = profile.KillCount;
            }
        }

        /// <summary>
        /// 检测异常行为
        /// </summary>
        private void DetectAnomalies(int playerId, PlayerBehaviorProfile profile)
        {
            var anomalies = new List<Anomaly>();

            // 1. 检测超高精准度
            if (profile.AverageAccuracy > 0.95f && profile.HitCount > 50)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.UnnaturalAccuracy,
                    Confidence = CalculateConfidence(profile.AverageAccuracy, 0.95f),
                    Details = $"精准度异常: {profile.AverageAccuracy:P2}"
                });
            }

            // 2. 检测超高爆头率
            if (profile.HeadshotRatio > 0.8f && profile.HitCount > 30)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.UnnaturalHeadshotRate,
                    Confidence = CalculateConfidence(profile.HeadshotRatio, 0.8f),
                    Details = $"爆头率异常: {profile.HeadshotRatio:P2}"
                });
            }

            // 3. 检测KD异常
            if (profile.KDRatio > 10.0f && profile.KillCount > 20)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.UnnaturalKDRatio,
                    Confidence = Math.Min(1.0f, profile.KDRatio / 20.0f),
                    Details = $"KD比异常: {profile.KDRatio:F2}"
                });
            }

            // 4. 检测击杀频率异常
            var recentKills = profile.RecentEvents.Count(e => e.Type == EventType.Kill);
            if (recentKills > 10 && recentKills > profile.KillCount * 0.5f)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.RapidKills,
                    Confidence = Math.Min(1.0f, recentKills / 20.0f),
                    Details = $"快速击杀: {recentKills} 次/{EventWindow}秒"
                });
            }

            // 5. 检测瞄准异常
            DetectAimAnomalies(profile, anomalies);

            // 6. 检测反应时间异常
            DetectReactionTimeAnomalies(profile, anomalies);

            // 处理检测到的异常
            foreach (var anomaly in anomalies)
            {
                if (anomaly.Confidence >= AnomalyThreshold)
                {
                    HandleAnomaly(playerId, anomaly);
                }
            }

            // 更新连续异常计数
            if (anomalies.Count > 0)
            {
                profile.ConsecutiveAnomalies++;
                if (profile.ConsecutiveAnomalies >= ConsecutiveAnomalyThreshold)
                {
                    OnSustainedAnomalies(playerId, profile.ConsecutiveAnomalies);
                }
            }
            else
            {
                profile.ConsecutiveAnomalies = 0;
            }

            profile.TotalAnomalies += anomalies.Count;
        }

        /// <summary>
        /// 检测瞄准异常
        /// </summary>
        private void DetectAimAnomalies(PlayerBehaviorProfile profile, List<Anomaly> anomalies)
        {
            // 分析瞄准角度变化
            var aimEvents = profile.RecentEvents.Where(e => e.Type == EventType.Aim).ToList();
            if (aimEvents.Count < 10) return;

            int snapCount = 0;
            int perfectTrackingCount = 0;

            for (int i = 1; i < aimEvents.Count; i++)
            {
                var current = aimEvents[i];
                var previous = aimEvents[i - 1];

                float angleDelta = Math.Abs(current.Value - previous.Value);
                double timeDelta = current.Timestamp - previous.Timestamp;

                // 检测瞬间转向（角度变化过大但时间过短）
                if (angleDelta > 90 && timeDelta < 0.05)
                {
                    snapCount++;
                }

                // 检测完美跟踪（角度变化恒定）
                if (i > 1)
                {
                    var beforePrevious = aimEvents[i - 2];
                    float prevDelta = Math.Abs(previous.Value - beforePrevious.Value);
                    if (Math.Abs(angleDelta - prevDelta) < 0.1f && angleDelta > 10)
                    {
                        perfectTrackingCount++;
                    }
                }
            }

            // 检测瞄准吸附
            if (snapCount > aimEvents.Count * 0.3f)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.AimSnapping,
                    Confidence = (float)snapCount / aimEvents.Count,
                    Details = $"瞄准吸附: {snapCount}/{aimEvents.Count}"
                });
            }

            // 检测机械瞄准
            if (perfectTrackingCount > aimEvents.Count * 0.4f)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.MechanicalAim,
                    Confidence = (float)perfectTrackingCount / aimEvents.Count,
                    Details = $"机械瞄准: {perfectTrackingCount}/{aimEvents.Count}"
                });
            }
        }

        /// <summary>
        /// 检测反应时间异常
        /// </summary>
        private void DetectReactionTimeAnomalies(PlayerBehaviorProfile profile, List<Anomaly> anomalies)
        {
            // 分析反应时间
            var reactionEvents = profile.RecentEvents.Where(e => e.Type == EventType.ReactionTime).ToList();
            if (reactionEvents.Count < 5) return;

            var reactionTimes = reactionEvents.Select(e => e.Value).ToList();
            float avgReactionTime = reactionTimes.Average();
            float minReactionTime = reactionTimes.Min();

            // 人类反应时间通常在150ms以上
            if (minReactionTime < 0.1f) // 100ms
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.InhumanReactionTime,
                    Confidence = Math.Min(1.0f, (0.1f - minReactionTime) * 10),
                    Details = $"反应时间异常: {minReactionTime * 1000:F0}ms"
                });
            }

            // 检测反应时间过于一致（机器特征）
            float variance = CalculateVariance(reactionTimes);
            if (variance < 0.001f && reactionTimes.Count > 10)
            {
                anomalies.Add(new Anomaly
                {
                    Type = AnomalyType.ConsistentReactionTime,
                    Confidence = 0.8f,
                    Details = $"反应时间过于一致: {avgReactionTime * 1000:F0}ms (方差: {variance:F6})"
                });
            }
        }

        /// <summary>
        /// 计算置信度
        /// </summary>
        private float CalculateConfidence(float value, float threshold)
        {
            if (value <= threshold) return 0;
            return Math.Min(1.0f, (value - threshold) / (1.0f - threshold));
        }

        /// <summary>
        /// 计算方差
        /// </summary>
        private float CalculateVariance(List<float> values)
        {
            if (values.Count < 2) return 0;
            float avg = values.Average();
            return values.Select(v => (v - avg) * (v - avg)).Average();
        }

        /// <summary>
        /// 处理异常
        /// </summary>
        private void HandleAnomaly(int playerId, Anomaly anomaly)
        {
            GD.PushWarning($"玩家 {playerId} 检测到异常 [{anomaly.Type}]: {anomaly.Details} " +
                          $"(置信度: {anomaly.Confidence:P})");

            EmitSignal(SignalName.AnomalyDetected, playerId, (int)anomaly.Type, anomaly.Confidence, anomaly.Details);
        }

        /// <summary>
        /// 处理持续异常
        /// </summary>
        private void OnSustainedAnomalies(int playerId, int count)
        {
            GD.PushError($"玩家 {playerId} 连续异常次数: {count}");
            EmitSignal(SignalName.SustainedAnomalies, playerId, count);
        }

        /// <summary>
        /// 获取玩家风险评分
        /// </summary>
        public float GetRiskScore(int playerId)
        {
            if (!_behaviorProfiles.TryGetValue(playerId, out var profile))
            {
                return 0;
            }

            // 计算综合风险评分 (0-100)
            float score = 0;

            // 基于异常数量
            score += Math.Min(30, profile.TotalAnomalies * 2);

            // 基于连续异常
            score += Math.Min(40, profile.ConsecutiveAnomalies * 8);

            // 基于行为统计数据
            if (profile.AverageAccuracy > 0.9f) score += 10;
            if (profile.HeadshotRatio > 0.7f) score += 10;
            if (profile.KDRatio > 5.0f) score += 10;

            return Math.Min(100, score);
        }

        /// <summary>
        /// 获取详细报告
        /// </summary>
        public BehaviorReport GetBehaviorReport(int playerId)
        {
            if (!_behaviorProfiles.TryGetValue(playerId, out var profile))
            {
                return null;
            }

            return new BehaviorReport
            {
                PlayerId = playerId,
                RiskScore = GetRiskScore(playerId),

                ConsecutiveAnomalies = profile.ConsecutiveAnomalies,
                AverageAccuracy = profile.AverageAccuracy,
                HeadshotRatio = profile.HeadshotRatio,
                KDRatio = profile.KDRatio,

                TotalDeaths = profile.DeathCount,

            };
        }

        /// <summary>

        }

        /// <summary>

        /// </summary>
        public void ResetPlayer(int playerId)

        }

        /// <summary>
        /// 记录违规行为
        /// </summary>
        private void RecordViolation(int playerId, SecurityViolation violation)
        {
            if (!_playerSecurityStates.TryGetValue(playerId, out var state))
            {
                return;
            }

            state.Violations.Add(new ViolationRecord
            {
                Type = violation,
                Timestamp = Time.GetUnixTimeFromSystem()
            });

            // 降低信任分
            int penalty = GetViolationPenalty(violation);
            state.TrustScore = Math.Max(0, state.TrustScore - penalty);

            GD.PushWarning($"玩家 {playerId} 违规 [{violation}], 信任分: {state.TrustScore}/{KickThreshold}");

            // 检查是否达到踢出阈值
            if (state.TrustScore <= 0)
            {
                EmitSignal(SignalName.PlayerShouldBeKicked, playerId, "信任分耗尽");
            }
        }

        /// <summary>
        /// 获取违规惩罚值
        /// </summary>
        private int GetViolationPenalty(SecurityViolation violation) => violation switch
        {
            SecurityViolation.SpeedHack => 30,
            SecurityViolation.Teleport => 40,
            SecurityViolation.InvalidPacket => 10,
            SecurityViolation.SequenceMismatch => 15,
            SecurityViolation.TimestampAnomaly => 20,
            SecurityViolation.AimBot => 50,
            SecurityViolation.WallHack => 50,
            _ => 5
        };

        /// <summary>
        /// 决定采取的行动
        /// </summary>
        private SecurityAction DetermineAction(int playerId)
        {
            if (!_playerSecurityStates.TryGetValue(playerId, out var state))
            {
                return SecurityAction.None;
            }

            // 检查最近的违规行为
            double now = Time.GetUnixTimeFromSystem();
            int recentViolations = 0;
            foreach (var violation in state.Violations)
            {
                if (now - violation.Timestamp < 60.0) // 最近60秒内
                {
                    recentViolations++;
                }
            }

            if (state.TrustScore <= 0)
            {
                return SecurityAction.Kick;
            }
            else if (recentViolations >= 5)
            {
                return SecurityAction.Warn;
            }
            else if (state.TrustScore < 30)
            {
                return SecurityAction.Spectate;
            }

            return SecurityAction.None;
        }

        {
            if (!_playerSecurityStates.TryGetValue(playerId, out var state))
            {
                return null;
            }

            var report = new SecurityReport
            {
                PlayerId = playerId,
                TrustScore = state.TrustScore,
                ConnectTime = state.ConnectTime,
                TotalViolations = state.Violations.Count,
                ViolationsByType = new Dictionary<SecurityViolation, int>()
            };

            foreach (var violation in state.Violations)
            {
                report.ViolationsByType[violation.Type] =
                    report.ViolationsByType.GetValueOrDefault(violation.Type, 0) + 1;
            }

            // 添加作弊检测报告
            if (CheatDetector != null)
            {
                report.CheatReport = CheatDetector.GetReport(playerId);
            }

            return report;
        }

15.5.1 安全设计原则

public class SpatialHash
{
    private Dictionary<int, List<int>> _cells = new();

    public void AddPlayer(int playerId, Vector3 position)
    {
        int cellId = GetCellId(position);
        if (!_cells.TryGetValue(cellId, out var list))
        {
            list = new List<int>();
            _cells[cellId] = list;
        }
        list.Add(playerId);
    }

    // 只检查同格和相邻格的玩家
    public List<int> GetNearbyPlayers(Vector3 position)
    {
        var nearby = new List<int>();
        int cellId = GetCellId(position);

        // 检查3x3x3区域
        for (int x = -1; x <= 1; x++)
        for (int y = -1; y <= 1; y++)
        for (int z = -1; z <= 1; z++)
        {
            int checkCell = cellId + x + y * 100 + z * 10000;
            if (_cells.TryGetValue(checkCell, out var players))
            {
                nearby.AddRange(players);
            }
        }

        return nearby;
    }
}
